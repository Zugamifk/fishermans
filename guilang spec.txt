GUI -> 'GUI' [ GUIHEADER ] [ WINDOWLIST ] ENDOFSTRING
GUIHEADER -> '(' GUIHEADERPARAMLIST ')'
GUIHEADERPARAMLIST -> GUIHEADERPARAM [ ',' GUIHEADERPARAMLIST ]
GUIHEADERPARAM -> ( POSITION | DIMENSION )
POSITION -> ( 'x' | 'y' ) ':' NUMBER
DIMENSION -> ( 'w' | 'h' ) ':' NUMBER
WINDOWLIST -> WINDOW [ WINDOWLIST ]
WINDOW -> 'WINDOW' [ WINDOWHEADER ] '{' [ CELL ] '}'
WINDOWHEADER -> '(' WINDOWHEADERPARAMLIST ')'
WINDOWHEADERPARAMLIST -> WINDOWHEADERPARAM [ ',' WINDOWHEADERPARAMLIST ] 
WINDOWHEADERPARAM -> ( POSITION | DIMENSION | WINDOWSTATE )
WINDOWSTATE -> 'state:' ( 'active' | 'inactive' )
CELL -> 'CELL' '{' ( CELLBRANCH | OBJECTARRAY ) '}'
CELLBRANCH -> CELLBRANCHHEADER CELLLIST
CELLLIST -> CELL [ CELLLIST ]
CELLBRANCHHEADER -> '(' ORIENTATION [ ',' PARTITIONS ] ')' 
ORIENTATION -> ( 'horizontal' | 'vertical' )
PARTITIONS -> NUMBER [ ',' PARTITIONS ]
