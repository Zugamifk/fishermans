SOUND -> @ '<' SEQUENCE '>'
SEQUENCE -> SIGNAL
SIGNAL -> '(' ( VALUE | ENVELOPE ) ')'
ENVELOPE -> ( LFO | ASDR ) 
VALUE -> ( ATOM | # | VARIABLE )
ATOM -> ( SIN | PRODUCT | SUM )
VARIABLE -> ( 'time' | FREQUENCY )

FREQUENCY -> 'freq' #

PRODUCT -> '*' SIGNAL SIGNAL
SUM -> '+' SIGNAL SIGNAL

SIN -> 'sin' SIGNAL

LFO -> 'lfo' SIGNAL SIGNAL
ASDR -> 'asdr' # # # # SIGNAL
